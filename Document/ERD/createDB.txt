CREATE TABLE exam_type (
	type_exam_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    exan_name VARCHAR(255) NOT NULL,
    link_exam VARCHAR(2000) NOT NULL,
	hide INT(1) NOT NULL
)
CREATE TABLE table_manager (
	table_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    type_exam_id INT NOT NULL,
    user_id INT NOT NULL,
    table_name VARCHAR(255) NOT NULL,
    number_question_use INT(3) NOT NULL,
	hide INT(1) NOT NULL,
    FOREIGN KEY (type_exam_id) REFERENCES exam_type(type_exam_id),
    FOREIGN KEY (user_id) REFERENCES user_info(user_id)
)
create table instruction (
	ins_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	instruction VARCHAR(2000) NOT NULL
)
create table code_submissions (
    code_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    table_id INT NOT NULL,
    code_content VARCHAR(10) NOT NULL,
	hide INT(1) NOT NULL,
    FOREIGN KEY (table_id) REFERENCES table_manager(table_id)
)
create table answer (
	answer_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	code_id INT NOT NULL,
    answer VARCHAR(2),
    FOREIGN KEY (code_id) REFERENCES code_submissions(code_id)
)
CREATE TABLE submissions (
	sub_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    table_id INT NOT NULL,
    code_id INT NOT NULL,
    number_ans_correct INT(3) NOT NULL,
    img_code VARCHAR(2000) NOT NULL,
	img_student_id VARCHAR(2000) NOT NULL,
    FOREIGN KEY (table_id) REFERENCES table_manager(table_id),
    FOREIGN KEY (code_id) REFERENCES code_submissions(code_id)
)
